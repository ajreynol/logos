def __smtx_typeof_value : SmtValue -> SmtType
  | (SmtValue.Boolean b) => SmtType.Bool
  | (SmtValue.Binary w n) =>
      if 0 <= w ∧ n = n % (2 ^ Int.toNat w) then

      else
        SmtType.None
  | (SmtValue.Map m) => (__smtx_typeof_map_value m)
  | (SmtValue.Seq ss) => (__smtx_typeof_seq_value ss)
  | (SmtValue.Apply f v) => (__smtx_typeof_apply_value (__smtx_typeof_value f) (__smtx_typeof_value v))
  | ... => ...

def __smtx_typeof_dt_cons_rec (T : SmtType) : SmtDatatype -> Nat -> SmtType
  | (SmtDatatype.sum SmtDatatypeCons.unit d), 0 => T
  | (SmtDatatype.sum (SmtDatatypeCons.cons U c) d), 0 =>
      (SmtType.DtcAppType U (__smtx_typeof_dt_cons_rec T (SmtDatatype.sum c d) 0))
  | (SmtDatatype.sum c d), (Nat.succ n) => (__smtx_typeof_dt_cons_rec T d n)
  | d, n => SmtType.None

def __smtx_ret_typeof_sel (s : String) (d : SmtDatatype) (n : Nat) (m : Nat) : SmtType :=
  (__smtx_ret_typeof_sel_rec (__smtx_dt_substitute s d d) n m)

def __smtx_typeof : SmtTerm -> SmtType
  | (SmtTerm.Boolean b) => SmtType.Bool
  | (SmtTerm.Binary w n) =>
      if 0 <= w ∧ n = n % (2 ^ Int.toNat w) then
        SmtType.BitVec (Int.toNat w)
      else
        SmtType.None
  | (SmtTerm.eq x1 x2) => (__smtx_typeof_eq (__smtx_typeof x1) (__smtx_typeof x2))
  | (SmtTerm.exists s T x1) =>
      if __smtx_typeof x1 = SmtType.Bool then SmtType.Bool else SmtType.None
  | (SmtTerm.plus x1 x2) => (__smtx_typeof_arith_overload_op_2 (__smtx_typeof x1) (__smtx_typeof x2))
  | (SmtTerm.select x1 x2) => (__smtx_typeof_select (__smtx_typeof x1) (__smtx_typeof x2))
  | (SmtTerm.Apply f x1) => (__smtx_typeof_apply (__smtx_typeof f) (__smtx_typeof x1))
  | (SmtTerm.DtCons s d i) =>
      let _v0 := (SmtType.Datatype s d)
      (__smtx_typeof_guard_wf _v0 (__smtx_typeof_dt_cons_rec _v0 (__smtx_dt_substitute s d d) i))
  | (SmtTerm.Apply (SmtTerm.DtSel s d i j) x1) =>
      let _v0 := (__smtx_ret_typeof_sel s d i j)
      (__smtx_typeof_guard_wf _v0 (__smtx_typeof_apply (SmtType.FunType (SmtType.Datatype s d) _v0) (__smtx_typeof x1)))
  | (SmtTerm.Apply (SmtTerm.DtTester s d i) x1) =>
      let _v0 := (SmtType.Datatype s d)
      (__smtx_typeof_guard (__smtx_typeof_dt_cons_rec _v0 (__smtx_dt_substitute s d d) i)
        (__smtx_typeof_apply (SmtType.FunType _v0 SmtType.Bool) (__smtx_typeof x1)))
  | ... => ...

def __smtx_type_default : SmtType -> SmtValue
  | SmtType.Bool => (SmtValue.Boolean false)
  | SmtType.Int => (SmtValue.Numeral 0)
  | SmtType.RegLan => (SmtValue.RegLan native_re_none)
  | (SmtType.Datatype s d) => (__smtx_datatype_default s d d 0)
  | (SmtType.FunType T U) => (SmtValue.Fun native_default_ifun_id T U)
  | ... => ...

def native_inhabited_type (T : SmtType) : Bool := by
  classical
  exact
    decide (__smtx_typeof_value (__smtx_type_default T) = T) &&
      (__smtx_value_canonical_bool (__smtx_type_default T))

def __smtx_dt_cons_wf_rec : SmtDatatypeCons -> RefList -> Bool
  | (SmtDatatypeCons.cons (SmtType.TypeRef s) c), refs =>
      if s ∈ refs then __smtx_dt_cons_wf_rec c refs else false
  | (SmtDatatypeCons.cons T c), refs =>
      if native_inhabited_type T then
        if __smtx_type_wf_rec T refs then __smtx_dt_cons_wf_rec c refs else false
      else
        false
  | SmtDatatypeCons.unit, refs => true

def __smtx_dt_wf_rec : SmtDatatype -> RefList -> Bool
  | SmtDatatype.null, refs => false
  | (SmtDatatype.sum c SmtDatatype.null), refs => (__smtx_dt_cons_wf_rec c refs)
  | (SmtDatatype.sum c d), refs =>
      if __smtx_dt_cons_wf_rec c refs then __smtx_dt_wf_rec d refs else false

def __smtx_type_wf_rec : SmtType -> RefList -> Bool
  | (SmtType.Datatype s d), refs =>
      if s ∈ refs then false else __smtx_dt_wf_rec d (s :: refs)
  | (SmtType.TypeRef s), refs => false
  | (SmtType.Seq x1), refs => (native_inhabited_type x1 && __smtx_type_wf_rec x1 [])
  | (SmtType.Map x1 x2), refs =>
      native_inhabited_type x1 &&
        __smtx_type_wf_rec x1 [] &&
        native_inhabited_type x2 &&
        __smtx_type_wf_rec x2 []
  | (SmtType.FunType x1 x2), refs => false
  | (SmtType.Set x1), refs => (native_inhabited_type x1 && __smtx_type_wf_rec x1 [])
  | (SmtType.DtcAppType x1 x2), refs => false
  | SmtType.None, refs => false
  | SmtType.RegLan, refs => false
  | T, refs => true

def __smtx_type_wf_component (T : SmtType) : Bool :=
  (native_inhabited_type T && __smtx_type_wf_rec T [])

def __smtx_type_wf : SmtType -> Bool
  | SmtType.RegLan => true
  | (SmtType.FunType T U) => (__smtx_type_wf_component T && __smtx_type_wf_component U)
  | T => (__smtx_type_wf_component T)
\end{lstlisting}

inductive smt_interprets : SmtModel -> SmtTerm -> Bool -> Prop
  | intro_true  (M : SmtModel) (t : SmtTerm) :

def native_fun_typed (M : SmtModel) : Prop :=
  ∀ fid A B i,
    __smtx_type_wf (SmtType.FunType A B) = true ->
    __smtx_typeof_value i = A ->
    __smtx_typeof_value (native_eval_ifun_apply M fid A B i) = B ∧

def model_total_typed (M : SmtModel) : Prop :=

  (∀ s T, __smtx_type_wf T = true -> __smtx_value_canonical_bool (native_model_lookup M s T)) = true ∧
  (∀ s T, __smtx_type_wf T = false -> native_model_lookup M s T = SmtValue.NotValue) ∧
  native_fun_typed M
